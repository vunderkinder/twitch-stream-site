        <input type="text" id="newChannel" placeholder="Введите имя канала">
        <button onclick="changeChannel()">Сменить канал</button>
    </div>

    <script>
        function changeChannel() {
            const channel = document.getElementById('newChannel').value.trim();
            if (channel) {
                document.getElementById('channelName').innerText = channel;
                const parentDomain = 'vunderkinder.github.io';
                document.getElementById('twitchPlayer').src = `https://player.twitch.tv/?channel=${channel}&parent=${parentDomain}`;
                document.getElementById('twitchChat').src = `https://www.twitch.tv/embed/${channel}/chat?parent=${parentDomain}`;
            }
        }
    </script>
